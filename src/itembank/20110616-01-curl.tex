  \begin{prob}
ベクトル場 $${\bf V}({\bf r})=({a}xy^3+{b}x^2,{c}x)$$ の渦度の, $${\bf r}=({x},{y})$$における値を求めよう.
  \end{prob}
  \begin{sol}
$$(\nabla\times{\bf V})_z=\frac{\partial}{\partial x} {c}x - \frac{\partial}{\partial y}({a}xy^3+{b}x^2)$$.
    \end{sol}
    \begin{ans}
$${c}-3*{a}*{x}*{y}*{y}$$      
    \end{ans}
